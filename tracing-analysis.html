Spec¶

  • hardware (per node): 2 x Intel Xeon E5-2620 v4 -- 8 cores/CPU -- 64 GiB -- 598 GB HDD -- 10 Gbps (SRIOV)
  • cluster: 5 nodes
  • clients: 3 * 32 threads
  • records: 30,000
  • operations: 300,000
  • tracing proportion: 1%
  • value sizes: 1, 2, 4, 8, ..., 2048 KB with a Zipf distribution (i.e., 1 KB is the most popular) ; alpha = 1.5
  • key popularity: Zipf distribution ; alpha = 0.99

In [11]:
summarise_measures <- function(f) {
    df.remote.measures %>%
        group_by(size_in_bytes) %>%
        summarise(user_request_waiting_time_in_ms = f(user_request_waiting_time_in_ms),
                  remote_request_waiting_time_in_ms = f(remote_request_waiting_time_in_ms),
                  remote_request_read_operation_in_ms = f(remote_request_read_operation_in_ms),
                  receive_response_time_in_ms = f(receive_response_time_in_ms), .groups = "drop") %>%
        pivot_longer(!size_in_bytes, names_to = "measure", values_to = "latency_in_ms")
}
